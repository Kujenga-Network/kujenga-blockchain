## Ubuntu Install

sudo apt-get update
sudo apt-get upgrade -y
sudo apt install git -y

git clone https://github.com/Kujenga-Network/kujenga-blockchain --recurse-submodules

#CLI:
cd kujenga-blockchain
sh install.sh
. ./activate
kujenga init

#GUI:
cd kujenga-blockchain
sh install.sh
. ./activate
sh install-gui.sh
cd kujenga-blockchain-gui
npm run electron &
